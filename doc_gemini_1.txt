// SeniorFriendlyGeminiUI.tsx
import React, { useState, useEffect, useRef } from "react";
import { w3cwebsocket as W3CWebSocket } from "websocket";


// ============================================================
// Types
// ============================================================
interface GeminiMessage {
  timestamp: string;
  role: "user" | "gemini";
  content: string;
  suggestedClips?: Array<{ title: string; url: string }>;
}


// ============================================================
// Component
// ============================================================
export const SeniorFriendlyGeminiUI: React.FC<{ userId: string }> = ({ userId }) => {
  const [messages, setMessages] = useState<GeminiMessage[]>([]);
  const [inputText, setInputText] = useState("");
  const videoRef = useRef<HTMLVideoElement>(null);


  // WebSocket for Gemini live conversation
  const clientRef = useRef<W3CWebSocket | null>(null);


  useEffect(() => {
    // Connect to your Gemini live server
    const client = new W3CWebSocket(`wss://your-gemini-server.com/live?userId=${userId}`);
    clientRef.current = client;


    client.onopen = () => console.log("[GeminiUI] Connected to live server");
    client.onmessage = (message) => {
      try {
        const data: GeminiMessage = JSON.parse(message.data as string);
        setMessages((prev) => [...prev, data]);
      } catch (err) {
        console.error(err);
      }
    };


    return () => client.close();
  }, [userId]);


  // ============================================================
  // Camera + Microphone
  // ============================================================
  useEffect(() => {
    async function setupCamera() {
      try {
        const stream = await navigator.mediaDevices.getUserMedia({ video: true, audio: true });
        if (videoRef.current) {
          videoRef.current.srcObject = stream;
        }
      } catch (err) {
        console.error("[GeminiUI] Camera/mic error:", err);
      }
    }
    setupCamera();
  }, []);


  // ============================================================
  // Send user input to Gemini
  // ============================================================
  const sendMessage = () => {
    if (!inputText.trim() || !clientRef.current) return;
    const msg: GeminiMessage = {
      timestamp: new Date().toISOString(),
      role: "user",
      content: inputText,
    };
    clientRef.current.send(JSON.stringify(msg));
    setMessages((prev) => [...prev, msg]);
    setInputText("");
  };


  // ============================================================
  // UI
  // ============================================================
  return (
    <div style={{ backgroundColor: "#1a1a1a", color: "#ffffff", fontSize: 24, padding: 20, height: "100vh", overflowY: "scroll" }}>
      <h1 style={{ color: "#ffcc00" }}>Gemini Live Assistant</h1>


      {/* Live Video */}
      <video
        ref={videoRef}
        autoPlay
        muted
        style={{ width: "50%", border: "4px solid #ffcc00", borderRadius: 10, marginBottom: 20 }}
      ></video>


      {/* Messages */}
      <div style={{ marginBottom: 20 }}>
        {messages.map((msg, idx) => (
          <div key={idx} style={{ marginBottom: 10 }}>
            <strong style={{ color: msg.role === "user" ? "#00ccff" : "#ffcc00" }}>
              {msg.role === "user" ? "You:" : "Gemini:"}
            </strong>{" "}
            <span>{msg.content}</span>
            {msg.suggestedClips && msg.suggestedClips.length > 0 && (
              <ul style={{ marginTop: 5, paddingLeft: 20 }}>
                {msg.suggestedClips.map((clip, i) => (
                  <li key={i}>
                    <a href={clip.url} target="_blank" rel="noopener noreferrer" style={{ color: "#00ff00" }}>
                      {clip.title}
                    </a>
                  </li>
                ))}
              </ul>
            )}
          </div>
        ))}
      </div>


      {/* Input */}
      <div style={{ display: "flex", gap: 10 }}>
        <input
          type="text"
          value={inputText}
          onChange={(e) => setInputText(e.target.value)}
          style={{ flex: 1, fontSize: 24, padding: 10, borderRadius: 5 }}
          placeholder="Ask Gemini..."
        />
        <button
          onClick={sendMessage}
          style={{ fontSize: 24, padding: "10px 20px", backgroundColor: "#ffcc00", border: "none", borderRadius: 5 }}
        >
          Send
        </button>
      </div>
    </div>
  );
};
